32static bool check_collision(const Game *game, int x, int y) {
33 // Check wall collision
34 if (x < 0 || x >= GRID_WIDTH || y < 0 || y >= GRID_HEIGHT) {
35 DEBUG_LOG("Wall collision at (%d, %d)", x, y);
36 return true;
37 }
38
39 // Check self collision

41 if (game->segments[i].x == x && game->segments[i].y == y) {
42 DEBUG_LOG("Self collision at (%d, %d)", x, y);
43 return true;
44 }
45 }

47 return false;

87static bool move_snake_internal(Game *game) {
88 if (!game || game->game_over)
89 return false;
90
91 // Calculate new head position
92 Position new_head = game->segments[0];
93 switch (game->direction) {
94 case UP:
95 new_head.y--;
96 break;
97 case RIGHT:
98 new_head.x++;
99 break;
100 case DOWN:
101 new_head.y++;
102 break;
103 case LEFT:
104 new_head.x--;
105 break;
106 }
107
108 DEBUG_LOG(
109 "Moving snake. Current head: (%d, %d), New head: (%d, %d), Direction: %d",
110 game->segments[0].x, game->segments[0].y, new_head.x, new_head.y,
111 game->direction);
112
113 // Check for collisions
114 if (check_collision(game, new_head.x, new_head.y)) {
115 game->game_over = true;
116 return false;
117 }
118
119 // Check for food collision
120 bool food_eaten = (new_head.x == game->food.x && new_head.y == game->food.y);
121 if (food_eaten) {
122 DEBUG_LOG("Food eaten at (%d, %d)", new_head.x, new_head.y);
123 }
124
125 // Move snake
126 if (food_eaten) {
127 // Grow snake
128 if (game->length >= game->capacity) {
129 game->capacity *= 2;
130 Position *new_segments =
131 realloc(game->segments, sizeof(Position) * game->capacity);
132 if (!new_segments) {
133 game->game_over = true;
134 return false;
135 }
136 game->segments = new_segments;
137 }
138
139 // Shift segments
140 for (int i = game->length; i > 0; i--) {
141 game->segments[i] = game->segments[i - 1];
142 }
143 game->length++;
144
145 // Update score
146 game->score += 10;
147
148 // Place new food
149 place_food(game);
150 } else {
151 // Shift segments
152 for (int i = game->length - 1; i > 0; i--) {
153 game->segments[i] = game->segments[i - 1];
154 }
155 }
156
157 // Update head
158 game->segments[0] = new_head;
159
160 return food_eaten;
161}
