Module snake_api

Python API for the Snake game C library.

This module provides a Python interface to the Snake game C library. It wraps the C functions and provides a more Pythonic interface for game control and state management.

Example

from src.api.snake_api import SnakeAPI

# Initialize the API with the path to the compiled C library
api = SnakeAPI("build/libsnake.so")

# Start playing
api.move()  # Move the snake
api.change_direction("right")  # Change direction
score = api.get_score()  # Get current score

class GameState(Structure):
    """Structure to hold game state data.
    
    This class represents the internal game state structure used by the C library.
    It maps directly to the C structure and is used for data exchange between
    Python and C code.
    
    Attributes:
        snake_positions: Pointer to array of snake segment positions
        snake_length: Current length of the snake
        food_position: Pointer to array containing food position
        score: Current game score
        is_game_over: Boolean indicating if game is over
    """

        ("snake_positions", POINTER(c_int)),
        ("snake_length", c_int),
        ("food_position", POINTER(c_int)),
        ("score", c_int),
        ("is_game_over", c_bool),
    ]

class SnakeAPI:
    """Python wrapper for the Snake game C library.
    
    This class provides a high-level interface to control the Snake game.
    It handles all communication with the C library and provides methods
    for game control, state management, and error handling.
    
    Attributes:
        lib: The loaded C library instance
        game_instance: Pointer to the current game instance in C
    
    Example:
        ```python
        api = SnakeAPI("build/libsnake.so")
        api.move()
        api.change_direction("right")
        state = api.get_state()
        ```
    """

    def __init__(self, lib_path: str):
        """Initialize the Snake API with the C library.

        Args:
            lib_path: Path to the compiled C library (.so file)

        Raises:
            RuntimeError: If the library file doesn't exist or game creation fails
        """
        if not os.path.exists(lib_path):
            raise RuntimeError(f"Library not found at {lib_path}")

        self.lib = CDLL(lib_path)
        self._setup_function_signatures()
        self.game_instance = self.lib.create_game()
        if not self.game_instance:
            raise RuntimeError("Failed to create game instance")

    def _setup_function_signatures(self):
        """Set up C function signatures for the library.
        
        This method configures the argument and return types for all C functions
        that will be called from Python. This is necessary for proper type
        conversion between Python and C.
        """
        # Game management
        self.lib.create_game.restype = c_void_p
        self.lib.destroy_game.argtypes = [c_void_p]
        self.lib.reset_game.argtypes = [c_void_p]

        # Movement and direction
        self.lib.move_snake.argtypes = [c_void_p]
        self.lib.move_snake.restype = c_bool
        self.lib.change_direction.argtypes = [c_void_p, c_char_p]
        self.lib.change_direction.restype = c_bool

        # State queries
        self.lib.is_game_over.argtypes = [c_void_p]
        self.lib.is_game_over.restype = c_bool
        self.lib.get_score.argtypes = [c_void_p]
        self.lib.get_score.restype = c_int
        self.lib.get_snake_length.argtypes = [c_void_p]
        self.lib.get_snake_length.restype = c_int

        # Position management
        self.lib.get_snake_positions.argtypes = [c_void_p, POINTER(c_int)]
        self.lib.get_food_position.argtypes = [c_void_p, POINTER(c_int)]
        self.lib.set_food_position.argtypes = [c_void_p, c_int, c_int]

    def cleanup(self):
        """Clean up resources used by the game instance.
        
        This method should be called when the game instance is no longer needed.
        It ensures proper cleanup of C resources to prevent memory leaks.
        """
        if hasattr(self, "game_instance") and self.game_instance:
            self.lib.destroy_game(self.game_instance)
            self.game_instance = None

    def __del__(self):
        """Destructor to ensure cleanup of resources.
        
        This method is called when the object is garbage collected.
        It ensures that C resources are properly cleaned up.
        """
        if hasattr(self, "game_instance") and self.game_instance:
            self.cleanup()

    def reset(self):
        """Reset the game to its initial state.
        
        This method resets all game state to initial values, including:
        - Snake position and length
        - Food position
        - Score
        - Game over status
        
        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        self.lib.reset_game(self.game_instance)

    def move(self) -> bool:
        """Move the snake in the current direction.

        This method advances the game by one step, moving the snake in its
        current direction. The snake will grow if it eats food, and the game
        will end if the snake hits a wall or itself.

        Returns:
            bool: True if the move was successful, False if game over

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        return self.lib.move_snake(self.game_instance)

    def change_direction(self, direction: str) -> bool:
        """Change the snake's direction.

        This method changes the direction in which the snake will move on
        the next step. The direction cannot be changed to the opposite of
        the current direction.

        Args:
            direction: One of "up", "right", "down", "left"

        Returns:
            bool: True if direction was changed successfully

        Raises:
            ValueError: If direction is invalid
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        if direction not in ["up", "right", "down", "left"]:
            raise ValueError(f"Invalid direction: {direction}")
        return self.lib.change_direction(self.game_instance, direction.encode())

    def is_game_over(self) -> bool:
        """Check if the game is over.

        The game is over if the snake has hit a wall or itself.

        Returns:
            bool: True if game is over, False otherwise

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        return self.lib.is_game_over(self.game_instance)

    def get_score(self) -> int:
        """Get the current game score.

        The score increases each time the snake eats food.

        Returns:
            int: Current score

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        return self.lib.get_score(self.game_instance)

    def get_snake_length(self) -> int:
        """Get the current length of the snake.

        The snake's length increases each time it eats food.

        Returns:
            int: Current snake length

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        return self.lib.get_snake_length(self.game_instance)

    def get_snake_positions(self) -> list[tuple[int, int]]:
        """Get the current positions of the snake's body segments.

        Returns a list of (x, y) coordinates representing the positions
        of all segments of the snake's body, with the head at index 0.

        Returns:
            list[tuple[int, int]]: List of (x, y) coordinates

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        length = self.get_snake_length()
        positions = (c_int * (length * 2))()
        self.lib.get_snake_positions(self.game_instance, positions)
        return [(positions[i], positions[i + 1]) for i in range(0, length * 2, 2)]

    def get_food_position(self) -> tuple[int, int]:
        """Get the current position of the food.

        Returns the (x, y) coordinates of the food item that the snake
        can eat to grow longer.

        Returns:
            tuple[int, int]: (x, y) coordinates of food

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        position = (c_int * 2)()
        self.lib.get_food_position(self.game_instance, position)
        return (position[0], position[1])

    def set_food_position(self, x: int, y: int):
        """Set the position of the food.

        This method allows manual placement of the food item at a specific
        position on the game board.

        Args:
            x: X coordinate
            y: Y coordinate

        Raises:
            RuntimeError: If the game instance is not initialized
        """
        if not self.game_instance:
            raise RuntimeError("Game instance not initialized")
        self.lib.set_food_position(self.game_instance, x, y)

    def get_state(self) -> dict:
        """Get the complete game state.

        Returns a dictionary containing all current game state information,
        including snake positions, food position, score, and game over status.

        Returns:
            dict: Game state containing:
                - snake: List of (x, y) coordinates for snake body
                - food: (x, y) coordinates of food
                - score: Current game score
                - game_over: Boolean indicating if game is over
        """
        return {
            "snake": self.get_snake_positions(),
            "food": self.get_food_position(),
            "score": self.get_score(),
            "game_over": self.is_game_over(),
        }
